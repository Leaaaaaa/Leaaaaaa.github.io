天涯网简单测试

发表于 2019-11-24

from selenium import webdriver
import time

# 实例化浏览器

driver = webdriver.Chrome()

# 打开项目URL

driver.get(‘https://bbs.tianya.cn/')

# 点击登录

driver.find_element_by_id(‘js_login’).click()

# 点击账号密码登录

driver.find_element_by_class_name(“normal-login-tab”).click()

# 输入账号

driver.find_element_by_id(“vwriter”).send_keys(“LeaLaaa”)

# 输入密码

driver.find_element_by_id(“vpassword”).send_keys(“***“)

# 点击登录
driver.find_element_by_class_name(“loginWin-submit-btn”).click()

time.sleep(5)

# 点击头像进入主页

driver.find_element_by_class_name(“logo”).click()

# 点击随机框进入随机发表编写

driver.find_element_by_xpath(“//
div[@class=’initText’]”).click()
time.sleep(5)

# 写随机标题

driver.find_element_by_css_selector(“#suiji_title”).send_keys(“这是一个测试”)

time.sleep(5)

# 写随机内容

driver.find_element_by_xpath(“//textarea[@id=’suiji_text’]”).send_keys(“this is my first web autotest”)
time.sleep(5)
# 点击发表

driver.find_element_by_xpath(“//input[@class=’submit’]”).click()

time.sleep(5)

driver.quit()
